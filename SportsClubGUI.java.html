import javax.swing.*;
import javax.swing.table.DefaultTableModel;
import java.awt.*;
import java.awt.event.*;
import java.sql.*;

public class SportsClubGUI {
    private final JFrame frame = new JFrame("Sports Club Management System");
    private final JTabbedPane tabbedPane = new JTabbedPane();
    private static final Color PRIMARY_COLOR = new Color(25, 118, 210),
            SECONDARY_COLOR = new Color(245, 245, 245),
            ACCENT_COLOR = new Color(46, 125, 50),
            TEXT_COLOR = new Color(33, 33, 33),
            BUTTON_HOVER_COLOR = new Color(21, 101, 192),
            TABLE_HEADER_COLOR = new Color(33, 150, 243);
    private static final String DB_URL = "jdbc:sqlserver://localhost:1433;databaseName=SPORTSCLUBDBMS;encrypt=true;trustServerCertificate=true",
            USER = "sportsclub_user", PASS = "SecurePassword123";

    public static void main(String[] args) {
        EventQueue.invokeLater(() -> {
            try {
                Class.forName("com.microsoft.sqlserver.jdbc.SQLServerDriver");
                new SportsClubGUI().frame.setVisible(true);
            } catch (Exception e) {
                JOptionPane.showMessageDialog(null, "Failed to load database driver: " + e.getMessage(),
                        "Driver Error", JOptionPane.ERROR_MESSAGE);
            }
        });
    }

    public SportsClubGUI() {
        frame.setBounds(100, 100, 900, 600);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.getContentPane().setBackground(SECONDARY_COLOR);
        tabbedPane.setBackground(SECONDARY_COLOR);
        tabbedPane.setForeground(TEXT_COLOR);
        tabbedPane.setFont(new Font("Segoe UI", Font.BOLD, 14));
        frame.add(tabbedPane);
        addPlayersTab();
        addClubsTab();
        addMatchesTab();
        addReportsTab();
    }

    private void addPlayersTab() {
        JPanel panel = createPanel();
        DefaultTableModel model = new DefaultTableModel() {
            public boolean isCellEditable(int row, int column) { return false; }
        };
        model.addColumn("ID"); model.addColumn("First Name"); model.addColumn("Last Name");
        model.addColumn("Email"); model.addColumn("Phone"); model.addColumn("Nationality"); model.addColumn("DOB");
        JTable table = styleTable(new JTable(model));
        JScrollPane scrollPane = createScrollPane(table);
        panel.add(scrollPane, BorderLayout.CENTER);
        JPanel buttonsPanel = new JPanel(new FlowLayout(FlowLayout.CENTER, 10, 10));
        buttonsPanel.setBackground(SECONDARY_COLOR);
        JButton addButton = createStyledButton("Add Player", ACCENT_COLOR),
                editButton = createStyledButton("Edit Player", PRIMARY_COLOR),
                deleteButton = createStyledButton("Delete Player", new Color(211, 47, 47)),
                refreshButton = createStyledButton("Refresh", PRIMARY_COLOR);
        buttonsPanel.add(addButton); buttonsPanel.add(editButton);
        buttonsPanel.add(deleteButton); buttonsPanel.add(refreshButton);
        panel.add(buttonsPanel, BorderLayout.SOUTH);
        refreshPlayerData(model);
        refreshButton.addActionListener(e -> refreshPlayerData(model));
        addButton.addActionListener(e -> {
            AddEditPlayerDialog dialog = new AddEditPlayerDialog(frame, "Add New Player", true, null);
            dialog.setVisible(true);
            if (dialog.isSaved()) refreshPlayerData(model);
        });
        editButton.addActionListener(e -> {
            int row = table.getSelectedRow();
            if (row == -1) {
                JOptionPane.showMessageDialog(frame, "Please select a player to edit",
                        "No Selection", JOptionPane.WARNING_MESSAGE);
                return;
            }
            AddEditPlayerDialog dialog = new AddEditPlayerDialog(frame, "Edit Player", true, (int) model.getValueAt(row, 0));
            dialog.setVisible(true);
            if (dialog.isSaved()) refreshPlayerData(model);
        });
        deleteButton.addActionListener(e -> {
            int row = table.getSelectedRow();
            if (row == -1) {
                JOptionPane.showMessageDialog(frame, "Please select a player to delete",
                        "No Selection", JOptionPane.WARNING_MESSAGE);
                return;
            }
            int id = (int) model.getValueAt(row, 0);
            if (JOptionPane.showConfirmDialog(frame, "Are you sure you want to delete " +
                            model.getValueAt(row, 1) + " " + model.getValueAt(row, 2) + "?", "Confirm Delete",
                    JOptionPane.YES_NO_OPTION) == JOptionPane.YES_OPTION) {
                try (Connection conn = DriverManager.getConnection(DB_URL, USER, PASS);
                     PreparedStatement pstmt = conn.prepareStatement("DELETE FROM sc_Player WHERE player_id=?")) {
                    pstmt.setInt(1, id);
                    pstmt.executeUpdate();
                    refreshPlayerData(model);
                } catch (SQLException ex) {
                    JOptionPane.showMessageDialog(frame, "Error deleting player: " + ex.getMessage(),
                            "Database Error", JOptionPane.ERROR_MESSAGE);
                }
            }
        });
        tabbedPane.addTab("Players", panel);
    }

    private void refreshPlayerData(DefaultTableModel model) {
        model.setRowCount(0);
        try (Connection conn = DriverManager.getConnection(DB_URL, USER, PASS);
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery("SELECT * FROM sc_Player ORDER BY last_name,first_name")) {
            while (rs.next()) {
                model.addRow(new Object[]{rs.getInt("player_id"), rs.getString("first_name"),
                        rs.getString("last_name"), rs.getString("email"), rs.getString("phone_number"),
                        rs.getString("nationality"), rs.getDate("dob")});
            }
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(frame, "Error loading players: " + e.getMessage(),
                    "Database Error", JOptionPane.ERROR_MESSAGE);
        }
    }

    class AddEditPlayerDialog extends JDialog {
        private boolean saved = false;
        private final Integer playerId;
        private final JTextField firstNameField = new JTextField(20), lastNameField = new JTextField(20),
                emailField = new JTextField(20), phoneField = new JTextField(20),
                nationalityField = new JTextField(20), dobField = new JTextField(20);

        public AddEditPlayerDialog(JFrame parent, String title, boolean modal, Integer playerId) {
            super(parent, title, modal);
            this.playerId = playerId;
            setSize(400, 350);
            setLocationRelativeTo(parent);
            setLayout(new GridLayout(7, 2, 10, 10));
            getContentPane().setBackground(SECONDARY_COLOR);
            styleTextField(firstNameField); styleTextField(lastNameField); styleTextField(emailField);
            styleTextField(phoneField); styleTextField(nationalityField); styleTextField(dobField);
            add(createStyledLabel("First Name:")); add(firstNameField);
            add(createStyledLabel("Last Name:")); add(lastNameField);
            add(createStyledLabel("Email:")); add(emailField);
            add(createStyledLabel("Phone:")); add(phoneField);
            add(createStyledLabel("Nationality:")); add(nationalityField);
            add(createStyledLabel("Date of Birth (YYYY-MM-DD):")); add(dobField);
            JButton saveButton = createStyledButton("Save", ACCENT_COLOR),
                    cancelButton = createStyledButton("Cancel", new Color(211, 47, 47));
            add(saveButton); add(cancelButton);
            if (playerId != null) loadPlayerData();
            saveButton.addActionListener(e -> savePlayer());
            cancelButton.addActionListener(e -> dispose());
        }

        private void loadPlayerData() {
            try (Connection conn = DriverManager.getConnection(DB_URL, USER, PASS);
                 PreparedStatement pstmt = conn.prepareStatement("SELECT * FROM sc_Player WHERE player_id=?")) {
                pstmt.setInt(1, playerId);
                ResultSet rs = pstmt.executeQuery();
                if (rs.next()) {
                    firstNameField.setText(rs.getString("first_name"));
                    lastNameField.setText(rs.getString("last_name"));
                    emailField.setText(rs.getString("email"));
                    phoneField.setText(rs.getString("phone_number"));
                    nationalityField.setText(rs.getString("nationality"));
                    dobField.setText(rs.getDate("dob").toString());
                }
            } catch (SQLException ex) {
                JOptionPane.showMessageDialog(this, "Error loading player data: " + ex.getMessage(),
                        "Database Error", JOptionPane.ERROR_MESSAGE);
            }
        }

        private void savePlayer() {
            if (firstNameField.getText().trim().isEmpty() || lastNameField.getText().trim().isEmpty() ||
                    emailField.getText().trim().isEmpty() || phoneField.getText().trim().isEmpty() ||
                    nationalityField.getText().trim().isEmpty() || dobField.getText().trim().isEmpty()) {
                JOptionPane.showMessageDialog(this, "All fields are required", "Validation Error", JOptionPane.ERROR_MESSAGE);
                return;
            }
            try {
                Date dob = Date.valueOf(dobField.getText());
                try (Connection conn = DriverManager.getConnection(DB_URL, USER, PASS)) {
                    String sql = playerId == null ? "INSERT INTO sc_Player (first_name,last_name,email,phone_number,nationality,dob) VALUES (?,?,?,?,?,?)"
                            : "UPDATE sc_Player SET first_name=?,last_name=?,email=?,phone_number=?,nationality=?,dob=? WHERE player_id=?";
                    try (PreparedStatement pstmt = conn.prepareStatement(sql)) {
                        pstmt.setString(1, firstNameField.getText());
                        pstmt.setString(2, lastNameField.getText());
                        pstmt.setString(3, emailField.getText());
                        pstmt.setString(4, phoneField.getText());
                        pstmt.setString(5, nationalityField.getText());
                        pstmt.setDate(6, dob);
                        if (playerId != null) pstmt.setInt(7, playerId);
                        pstmt.executeUpdate();
                        saved = true;
                        dispose();
                    }
                }
            } catch (IllegalArgumentException e) {
                JOptionPane.showMessageDialog(this, "Invalid date format. Please use YYYY-MM-DD",
                        "Validation Error", JOptionPane.ERROR_MESSAGE);
            } catch (SQLException e) {
                JOptionPane.showMessageDialog(this, "Error saving player: " + e.getMessage(),
                        "Database Error", JOptionPane.ERROR_MESSAGE);
            }
        }

        public boolean isSaved() { return saved; }
    }

    private void addClubsTab() {
        JPanel panel = createPanel();
        DefaultTableModel model = new DefaultTableModel();
        model.addColumn("ID"); model.addColumn("Club Name"); model.addColumn("Location");
        JTable table = styleTable(new JTable(model));
        panel.add(createScrollPane(table), BorderLayout.CENTER);

    }

    private void refreshClubsData(DefaultTableModel model) {
        model.setRowCount(0);
        try (Connection conn = DriverManager.getConnection(DB_URL, USER, PASS);
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery("SELECT * FROM sc_Club ORDER BY club_name")) {
            while (rs.next()) {
                model.addRow(new Object[]{rs.getInt("club_id"), rs.getString("club_name"), rs.getString("location")});
            }
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(frame, "Error loading clubs: " + e.getMessage(),
                    "Database Error", JOptionPane.ERROR_MESSAGE);
        }
    }

    private void addMatchesTab() {
        JPanel panel = createPanel();
        DefaultTableModel model = new DefaultTableModel();
        model.addColumn("Match ID"); model.addColumn("Date"); model.addColumn("Home Team");
        model.addColumn("Away Team"); model.addColumn("Result"); model.addColumn("Venue");
        JTable table = styleTable(new JTable(model));
        panel.add(createScrollPane(table), BorderLayout.CENTER);
        refreshMatchesData(model);
        tabbedPane.addTab("Matches", panel);
    }

    private void refreshMatchesData(DefaultTableModel model) {
        model.setRowCount(0);
        try (Connection conn = DriverManager.getConnection(DB_URL, USER, PASS);
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery("SELECT m.match_id,m.match_date,ht.team_name as home_team," +
                     "at.team_name as away_team,m.result,v.venue_name FROM sc_Matches m JOIN sc_Team ht ON m.home_team_id=ht.team_id " +
                     "JOIN sc_Team at ON m.away_team_id=at.team_id JOIN sc_Venue v ON m.venue_id=v.venue_id ORDER BY m.match_date DESC")) {
            while (rs.next()) {
                model.addRow(new Object[]{rs.getInt("match_id"), rs.getDate("match_date"), rs.getString("home_team"),
                        rs.getString("away_team"), rs.getString("result"), rs.getString("venue_name")});
            }
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(frame, "Error loading matches: " + e.getMessage(),
                    "Database Error", JOptionPane.ERROR_MESSAGE);
        }
    }

    private JPanel createPanel() {
        JPanel panel = new JPanel(new BorderLayout());
        panel.setBackground(SECONDARY_COLOR);
        panel.setBorder(BorderFactory.createEmptyBorder(10, 10, 10, 10));
        return panel;
    }

    private JScrollPane createScrollPane(JComponent component) {
        JScrollPane scrollPane = new JScrollPane(component);
        scrollPane.getViewport().setBackground(SECONDARY_COLOR);
        scrollPane.setBorder(BorderFactory.createLineBorder(new Color(200, 200, 200)));
        return scrollPane;
    }

    private JTable styleTable(JTable table) {
        table.setBackground(SECONDARY_COLOR);
        table.setForeground(TEXT_COLOR);
        table.setGridColor(new Color(200, 200, 200));
        table.setFont(new Font("Segoe UI", Font.PLAIN, 13));
        table.setRowHeight(25);
        table.getTableHeader().setBackground(TABLE_HEADER_COLOR);
        table.getTableHeader().setForeground(Color.WHITE);
        table.getTableHeader().setFont(new Font("Segoe UI", Font.BOLD, 13));
        table.setSelectionBackground(new Color(187, 222, 251));
        table.setSelectionForeground(TEXT_COLOR);
        return table;
    }

    private JButton createStyledButton(String text, Color bgColor) {
        JButton button = new JButton(text) {
            protected void paintComponent(Graphics g) {
                Graphics2D g2 = (Graphics2D) g.create();
                g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
                g2.setPaint(new GradientPaint(0, 0, bgColor, 0, getHeight(),
                        new Color(Math.max(0, bgColor.getRed() - 20), Math.max(0, bgColor.getGreen() - 20),
                                Math.max(0, bgColor.getBlue() - 20))));
                g2.fillRoundRect(0, 0, getWidth(), getHeight(), 10, 10);
                super.paintComponent(g);
                g2.dispose();
            }
        };
        button.setBackground(bgColor);
        button.setForeground(Color.WHITE);
        button.setFont(new Font("Segoe UI", Font.BOLD, 13));
        button.setFocusPainted(false);
        button.setBorder(BorderFactory.createEmptyBorder(10, 20, 10, 20));
        button.setContentAreaFilled(false);
        button.addMouseListener(new MouseAdapter() {
            public void mouseEntered(MouseEvent evt) { button.setBackground(BUTTON_HOVER_COLOR); }
            public void mouseExited(MouseEvent evt) { button.setBackground(bgColor); }
        });
        return button;
    }

    private void styleTextField(JTextField textField) {
        textField.setBackground(SECONDARY_COLOR);
        textField.setForeground(TEXT_COLOR);
        textField.setFont(new Font("Segoe UI", Font.PLAIN, 13));
        textField.setBorder(BorderFactory.createCompoundBorder(
                BorderFactory.createLineBorder(new Color(200, 200, 200)),
                BorderFactory.createEmptyBorder(5, 5, 5, 5)));
        textField.setOpaque(true);
    }

    private JLabel createStyledLabel(String text) {
        JLabel label = new JLabel(text);
        label.setForeground(TEXT_COLOR);
        label.setFont(new Font("Segoe UI", Font.BOLD, 13));
        return label;
    }
